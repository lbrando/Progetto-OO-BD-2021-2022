CREATE TABLE IF NOT EXISTS public."Tematica"
(
    "Tematica" "char"[] NOT NULL,
    "ID_Corso" integer NOT NULL,
    "Keywords" "char"[],
    CONSTRAINT "FK_Tematica" FOREIGN KEY ("ID_Corso")
        REFERENCES public."Corso" ("ID_Corso") MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
)

TABLESPACE pg_default;
